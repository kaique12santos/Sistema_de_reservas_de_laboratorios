-- ==============================================================================
-- SISTEMA DE RESERVA DE LABORATÓRIOS - FATEC ZL
-- Versão do Schema: 2.1 (Revisão de Metadados e Encoding)
-- Banco de Dados: MySQL
-- ==============================================================================

-- 1. Criação do Banco de Dados com suporte a UTF8MB4
DROP DATABASE IF EXISTS fatecZL_lab_reservations;
CREATE DATABASE fatecZL_lab_reservations
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_unicode_ci;

USE fatecZL_lab_reservations;

-- ==============================================================================
-- 1. Tabelas Auxiliares e Configurações (Lookups)
-- ==============================================================================

CREATE TABLE departments (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL UNIQUE COMMENT 'Nome completo do curso ou departamento',
    code VARCHAR(10) NULL COMMENT 'Sigla (ex: DSM, GTI, GE)',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB COMMENT='Catálogo de cursos e departamentos para padronização';

-- ==============================================================================
-- 2. Tabelas de Acesso e Usuários
-- ==============================================================================

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL COMMENT 'Nome completo do usuário',
    email VARCHAR(150) NOT NULL UNIQUE COMMENT 'Email institucional, usado para login',
    password_hash VARCHAR(255) NOT NULL COMMENT 'Senha criptografada (BCrypt/Argon2)',
    password_reset_token VARCHAR(255) NULL COMMENT 'Token único para reset de senha',
    password_reset_expires TIMESTAMP NULL COMMENT 'Validade do token de reset',
    verification_token VARCHAR(255) NULL COMMENT 'Token para confirmar posse do e-mail (opcional)',
    department VARCHAR(255) NULL COMMENT 'Nome do departamento',
    -- Role define as permissões macro (Baseado no arquivo constants/roles.js)
    role ENUM('ADMIN', 'PROFESSOR', 'SUPPORT') NOT NULL DEFAULT 'PROFESSOR' COMMENT 'Nível de acesso',
    
    department_id INT NULL,
    status ENUM('PENDING', 'APPROVED', 'REJECTED', 'ACTIVE') NOT NULL DEFAULT 'PENDING' COMMENT 'Fluxo de aprovação de cadastro',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (department_id) REFERENCES departments(id) ON DELETE SET NULL,
    INDEX idx_user_email (email),
    INDEX idx_user_status (status)
) ENGINE=InnoDB COMMENT='Tabela central de usuários e autenticação';

-- ==============================================================================
-- 3. Tabelas de Configuração do Sistema (Ciclos e Ambientes)
-- ==============================================================================

CREATE TABLE laboratories (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50) NOT NULL COMMENT 'Identificação da sala (ex: Lab 01)',
    location VARCHAR(100) NULL COMMENT 'Localização física (ex: Bloco A)',
    capacity INT NOT NULL DEFAULT 0 COMMENT 'Capacidade máxima de alunos',
    description_lab TEXT NULL COMMENT 'Recursos disponíveis (Projetores, Softwares, etc)',
    is_active BOOLEAN DEFAULT TRUE COMMENT 'Soft delete: define se o lab está disponível para novas reservas',
    type ENUM('LABORATORIO', 'SALA DE AULA', 'AUDITORIO') NOT NULL DEFAULT 'LABORATORIO',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB COMMENT='Cadastro de espaços físicos reserváveis';

CREATE TABLE academic_cycles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(20) NOT NULL COMMENT 'Identificador do semestre (ex: 2026-1)',
    start_date DATE NOT NULL COMMENT 'Início do período letivo',
    end_date DATE NOT NULL COMMENT 'Fim do período letivo',
    admin_exclusive_end_date DATE NULL COMMENT 'Data limite para prioridade de reserva dos coordenadores',
    is_active BOOLEAN DEFAULT FALSE COMMENT 'Indica se este é o ciclo atual do sistema',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB COMMENT='Definição de semestres e janelas de reserva';

CREATE TABLE holidays (
    id INT AUTO_INCREMENT PRIMARY KEY,
    cycle_id INT NOT NULL COMMENT 'Vínculo com o ciclo acadêmico',
    date DATE NOT NULL COMMENT 'Data do bloqueio',
    description VARCHAR(100) NOT NULL COMMENT 'Motivo do feriado ou recesso',
    FOREIGN KEY (cycle_id) REFERENCES academic_cycles(id) ON DELETE CASCADE,
    UNIQUE KEY unique_holiday_date (date)
) ENGINE=InnoDB COMMENT='Datas onde não são permitidas reservas';

CREATE TABLE time_slots (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(20) NOT NULL COMMENT 'Identificador do horário (ex: M1, N2)',
    start_time TIME NOT NULL,
    end_time TIME NOT NULL,
    is_active BOOLEAN DEFAULT TRUE,
    INDEX idx_time_order (start_time)
) ENGINE=InnoDB COMMENT='Grade de horários padrão da FATEC';

-- ==============================================================================
-- 4. Tabelas de Reserva (Core Business)
-- ==============================================================================

CREATE TABLE reservations (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL COMMENT 'Professor solicitante',
    lab_id INT NOT NULL COMMENT 'Laboratório solicitado',
    cycle_id INT NOT NULL COMMENT 'Ciclo referente à reserva',
    type ENUM('SINGLE', 'RECURRING') NOT NULL DEFAULT 'SINGLE' COMMENT 'Reserva pontual ou para o semestre todo',
    status ENUM('PENDING', 'APPROVED', 'REJECTED', 'CANCELED') NOT NULL DEFAULT 'PENDING',
    reason TEXT NULL COMMENT 'Justificativa para rejeição ou observações',
    approved_by INT NULL COMMENT 'ID do administrador que processou o pedido',
    approval_date TIMESTAMP NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE RESTRICT,
    FOREIGN KEY (lab_id) REFERENCES laboratories(id) ON DELETE RESTRICT,
    FOREIGN KEY (cycle_id) REFERENCES academic_cycles(id) ON DELETE RESTRICT,
    FOREIGN KEY (approved_by) REFERENCES users(id) ON DELETE SET NULL
) ENGINE=InnoDB COMMENT='Cabeçalho do pedido de reserva';

CREATE TABLE reservation_items (
    id INT AUTO_INCREMENT PRIMARY KEY,
    reservation_id INT NOT NULL,
    lab_id INT NOT NULL,
    date DATE NOT NULL COMMENT 'Data específica da aula/uso',
    time_slot_id INT NULL,
    start_time TIME NOT NULL COMMENT 'Cópia denormalizada para verificação rápida de conflito',
    end_time TIME NOT NULL,
    note VARCHAR(255) NULL COMMENT 'Observação para este dia específico',
    status ENUM('ACTIVE', 'CANCELED', 'OVERWRITTEN') NOT NULL DEFAULT 'ACTIVE',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (reservation_id) REFERENCES reservations(id) ON DELETE CASCADE,
    FOREIGN KEY (lab_id) REFERENCES laboratories(id) ON DELETE CASCADE,
    FOREIGN KEY (time_slot_id) REFERENCES time_slots(id) ON DELETE SET NULL,
    -- CONSTRAINT DE INTEGRIDADE: Impede conflito direto no banco
    UNIQUE KEY unique_conflict_check (lab_id, date, start_time, status)
) ENGINE=InnoDB COMMENT='Instâncias individuais de reserva (os dias de aula de fato)';

-- ==============================================================================
-- 5. Auditoria e Logs
-- ==============================================================================

CREATE TABLE audit_logs (
    id INT AUTO_INCREMENT PRIMARY KEY,
    action ENUM('CREATE', 'UPDATE', 'DELETE', 'OVERWRITE', 'APPROVE', 'REJECT') NOT NULL,
    table_name VARCHAR(50) NOT NULL,
    record_id INT NOT NULL,
    changed_by INT NOT NULL,
    old_values JSON NULL COMMENT 'Estado anterior dos dados em formato JSON',
    new_values JSON NULL COMMENT 'Novo estado dos dados em formato JSON',
    timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (changed_by) REFERENCES users(id)
) ENGINE=InnoDB COMMENT='Logs de segurança e rastreabilidade de alterações';
