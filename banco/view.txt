USE fatecZL_lab_reservations;

CREATE OR REPLACE VIEW view_calendario_reservas AS
SELECT 
    ri.date AS data_aula, 
    ts.name AS horario, 
    l.name AS lab, 
    u.name AS professor, 
    d.code AS depto
FROM reservation_items ri
JOIN time_slots ts ON ri.time_slot_id = ts.id
JOIN laboratories l ON ri.lab_id = l.id
JOIN reservations r ON ri.reservation_id = r.id
JOIN users u ON r.user_id = u.id
LEFT JOIN departments d ON u.department_id = d.id
WHERE ri.is_active = 1  -- Mudança aqui: de 'ACTIVE' para 1
  AND u.is_active = 1;  -- Garante que o professor não foi desativado
